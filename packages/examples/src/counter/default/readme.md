```tsx [script]
import { Element, State } from '@htmlplus/element';

@Element()
class CounterDefault {
  @State()
  play = false;
  render() {
    return (
      <div class="center">
        <plus-counter to="1000" play={this.play} onPlusComplete={() => this.play = false}></plus-counter>
        <br />
        <button onClick={() => this.play = true}>Start</button>
      </div>
    );
  }
}
```
